\begin{defn}
  Consider a set $\{a,b\}$ consisting of two objects $a$ and $b$. We say that the
  \textit{ordered pair} $(a,b)$ is the set
  \begin{align*}
    \{\{a\},\{a,b\}\}.
  \end{align*}
  $a$ is then called the \textit{first element} of $(a,b)$, and $b$ is called the
  \textit{second element}.
\end{defn}
